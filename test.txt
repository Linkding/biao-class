function set_form_data(form, data) {
    // 遍历对象
    for (var key in data) {
        var value = data[key];
        //找到当前属性在表单中的位置
        var input = document.querySelector(`[name=${key}]`)
        //如果不存在，则继续下一个
        console.log(input)
        if (!input)
            continue;
            // console.log(input)
        var input_type = typeof value;
        switch (input_type) {
            /*如果是字符串或者数字，就默认其为input[type=number|text|url|...]*/
            case 'string':
            case 'number':
                input.value = value;
                break;
            /*如果是布尔值，就默认其为input[type=radio|checkbox]*/
            case 'boolean':
                input.checked = value;
                break;
        }
    }
}
